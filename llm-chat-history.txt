Got it. The executable (voidsend) is not recognized globally because cargo install places binaries in $HOME/.cargo/bin, which is not automatically added to your $PATH in Termux (especially with the Rust package from pkg install rust).
Quick Fix (Recommended)
Run these commands one by one:
# 1. Add Cargo bin directory to PATH permanently
echo 'export PATH="$HOME/.cargo/bin:$PATH"' >> \~/.bashrc

# 2. Reload the shell
source \~/.bashrc

# 3. Re-install the binary using your dev helper
cdv
dev install
Now test it:
voidsend --help
If it works, you're good.
Alternative: Manual Copy Method (Faster for Development)
If you prefer not to use cargo install every time, you can copy the built binary directly to Termux's PATH ($PREFIX/bin):
cdv
dev release          # builds optimized small binary
cp target/release/voidsend $PREFIX/bin/
voidsend --help
This method is often more reliable on Termux and doesn't rely on .cargo/bin.

#!/data/data/com.termux/files/usr/bin/bash
# VoidSend Dev Helper - Streamlined for Revvl Tab 2 + Termux

case "$1" in
  build)
    echo "→ Fast debug build..."
    cargo build --jobs 2
    ;;

  release|vr)
    echo "→ Safe optimized release build..."
    cargo build --release --jobs 1
    strip target/release/voidsend 2>/dev/null || true
    ls -lh target/release/voidsend
    ;;

  install|vi)
    echo "→ Installing binary to Termux PATH ($PREFIX/bin)..."
    cargo build --release --jobs 1
    strip target/release/voidsend 2>/dev/null || true
    cp target/release/voidsend $PREFIX/bin/voidsend
    chmod +x $PREFIX/bin/voidsend
    echo "→ voidsend is now available globally!"
    ;;

  check|vc)
    cargo check --jobs 2
    ;;

  run)
    cargo run -- --help
    ;;

  clean)
    cargo clean
    ;;

  status|vs)
    git status -s
    git log --oneline -1
    ;;

  push)
    if [ -z "$2" ]; then
      echo "Usage: ./dev.sh push 'commit message'"
      exit 1
    fi
    git add .
    git commit -m "$2" || true
    git push
    ;;

  *)
    echo "VoidSend Quick Commands:"
    echo "  dev install     → Build + copy to $PREFIX/bin (global)"
    echo "  dev release     → Optimized build"
    echo "  dev check       → Quick check"
    echo "  voidsend --help → Test the binary"
    echo ""
    echo "Aliases: vb, vr, vi, vc, vs, dev, cdv"
    ;;
esac
